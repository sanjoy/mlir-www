25 /// Bufferization of arith.constant. Replace with memref.get_global.
26 struct ConstantOpInterface
27  : public BufferizableOpInterface::ExternalModel<ConstantOpInterface,
28  arith::ConstantOp> {
29  LogicalResult bufferize(Operation *op, RewriterBase &rewriter,
30  const BufferizationState &state) const {
31  auto constantOp = cast<arith::ConstantOp>(op);
32 
33  // Only ranked tensors are supported.
34  if (!constantOp.getType().isa<RankedTensorType>())
35  return failure();
36 
37  // Only constants inside a module are supported.
38  auto moduleOp = constantOp->getParentOfType<ModuleOp>();
39  if (!moduleOp)
40  return failure();
41 
42  // Create global memory segment and replace tensor with memref pointing to
43  // that memory segment.
44  GlobalCreator globalCreator(moduleOp);
45  auto globalMemref = globalCreator.getGlobalFor(constantOp);
46  replaceOpWithNewBufferizedOp<memref::GetGlobalOp>(
47  rewriter, op, globalMemref.type(), globalMemref.getName());
48 
49  return success();
50  }
51 
52  bool isWritable(Operation *op, Value value,
53  const BufferizationState &state) const {
54  // Memory locations returned by memref::GetGlobalOp may not be written to.
55  assert(value.isa<OpResult>());
56  return false;
57  }
58 };
