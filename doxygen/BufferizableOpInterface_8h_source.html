58  /// Allow-list the given dialects in the dialect filter. Only ops from
59  /// allow-listed dialects will be bufferized. If no dialect is added, ops from
60  /// any dialect will be bufferized.
61  template <typename... DialectTs>
62  void addToDialectFilter() {
63  // The following expands a call to addToDialectFilterImpl for each dialect
64  // in 'DialectTs'. This magic is necessary due to a limitation in the places
65  // that a parameter pack can be expanded in c++11.
66  // FIXME: In c++17 this can be simplified by using 'fold expressions'.
67  (void)std::initializer_list<int>{
68  0, (addToDialectFilterImpl<DialectTs>(), 0)...};
69  }

113  /// Only bufferize ops from dialects that are allowed-listed by the filter.
114  /// All other ops are ignored. This option controls the scope of partial

127  if (!dialectFilter.hasValue())
128  dialectFilter.emplace();
129  dialectFilter->insert(DialectT::getDialectNamespace());

156 /// BufferizationState provides a variety of helper functions for dealing with
157 /// tensor values and memref buffers.
158 class BufferizationState {
159 public:
160  /// Determine which OpOperand* will alias with `result` if the op is
161  /// bufferized in place. Return an empty vector if the op is not bufferizable.
162  SmallVector<OpOperand *> getAliasingOpOperand(OpResult result) const;
163 
164  /// Determine which OpResult will alias with `opOperand` if the op is
165  /// bufferized in place. Return an empty OpResult if the op is not
166  /// bufferizable.
167  OpResult getAliasingOpResult(OpOperand &opOperand) const;
168 
169  /// Return true if `opOperand` bufferizes to a memory read. Return `true` if
170  /// the op is not bufferizable.
171  bool bufferizesToMemoryRead(OpOperand &opOperand) const;
172 
173  /// Return true if `opOperand` bufferizes to a memory write. Return true` if
174  /// the op is not bufferizable.
175  bool bufferizesToMemoryWrite(OpOperand &opOperand) const;
176 
177  /// Return true if `opOperand` does neither read nor write but bufferizes to
178  /// an alias. Return false if the op is not bufferizable.
179  bool bufferizesToAliasOnly(OpOperand &opOperand) const;
180 
181  /// Return true if the given value is read by an op that bufferizes to a
182  /// memory read. Also takes into account ops that create an alias but do not
183  /// read by themselves (e.g., ExtractSliceOp).
184  bool isValueRead(Value value) const;
185 
186  /// Starting from `value`, follow the use-def chain in reverse, always
187  /// selecting the aliasing OpOperands. Find and return Values for which
188  /// `condition` evaluates to true. OpOperands of such matching Values are not
189  /// traversed any further.
190  ///
191  /// When reaching the end of a chain (BlockArgument or Value without aliasing
192  /// OpOperands), also return the last Value of that chain.
193  ///
194  /// Example:
195  ///
196  /// 8
197  /// |
198  /// 6* 7* +-----+----+
199  /// | | | |
200  /// 2* 3 4* 5
201  /// | | | |
202  /// +----------+----------+----------+
203  /// |
204  /// 1
205  ///
206  /// In the above example, Values with a star satisfy the condition. When
207  /// starting the traversal from Value 1, the resulting SetVector is:
208  /// { 2, 7, 8, 5 }
209  SetVector<Value> findValueInReverseUseDefChain(
210  Value value, llvm::function_ref<bool(Value)> condition) const;
211 
212  /// Find the Values of the last preceding write of a given Value.
213  ///
214  /// Note: Unknown ops are handled conservatively and assumed to be writes.
215  /// Furthermore, BlockArguments are also assumed to be writes. There is no
216  /// analysis across block boundaries.
217  ///
218  /// Note: When reaching an end of the reverse SSA use-def chain, that value
219  /// is returned regardless of whether it is a memory write or not.
220  SetVector<Value> findLastPrecedingWrite(Value value) const;
221 
222  /// Return `true` if the given OpResult has been decided to bufferize inplace.
223  virtual bool isInPlace(OpOperand &opOperand) const = 0;
224 
225  /// Return true if `v1` and `v2` bufferize to equivalent buffers.
226  virtual bool areEquivalentBufferizedValues(Value v1, Value v2) const = 0;
227 
228  /// Return the buffer (memref) for a given OpOperand (tensor). Allocate
229  /// a new buffer and copy over data from the existing buffer if out-of-place
230  /// bufferization was decided.
231  FailureOr<Value>
232  getBuffer(RewriterBase &rewriter, OpOperand &opOperand,
233  bool forceInPlace = false,
234  Optional<Operation *> customCopyInsertionPoint = None) const;
235 
236  /// Return dialect-specific bufferization state.
237  template <typename StateT>
238  Optional<const StateT *> getDialectState(StringRef name) const {
239  auto it = dialectState.find(name);
240  if (it == dialectState.end())
241  return None;
242  return static_cast<const StateT *>(it->getSecond().get());
243  }
244 
245  /// Return dialect-specific bufferization state or create one if none exists.
246  template <typename StateT>
247  StateT &getOrCreateDialectState(StringRef name) {
248  // Create state if it does not exist yet.
249  if (!dialectState.count(name))
250  dialectState[name] = std::make_unique<StateT>();
251  return static_cast<StateT &>(*dialectState[name]);
252  }
253 
254  /// Return a reference to the BufferizationOptions.
255  const BufferizationOptions &getOptions() const { return options; }
256 
257 protected:
258  BufferizationState(const BufferizationOptions &options);
259 
260  // BufferizationState should be passed as a reference.
261  BufferizationState(const BufferizationState &) = delete;
262 
263  ~BufferizationState() = default;
264 
265 private:
266  /// Dialect-specific bufferization state.
267  DenseMap<StringRef, std::unique_ptr<DialectBufferizationState>> dialectState;
268 
269  /// A reference to current bufferization options.
270  const BufferizationOptions &options;
271 };

339 /// AllocationHoistingBarrierOnly is an external implementation of

342 /// `isAllocationHoistingBarrier`) are implemented conservatively.

348  const BufferizationState &state) const {
349  return true;

354  return true;
355  }

359  const BufferizationState &state) const {
360  return {};
361  }

364  const BufferizationState &state) const {
365  return OpResult();
