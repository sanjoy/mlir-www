190  /// Append a pattern to tile the Op `opName` and fuse its producers with
191  /// tiling and fusion `options`.
192  CodegenStrategy &
193  tileAndFuse(StringRef opName, const LinalgTilingAndFusionOptions &options,
194  const LinalgTransformationFilter::FilterFunction &f = nullptr) {
195  transformationSequence.emplace_back(
196  std::make_unique<TileAndFuse>(opName, options, f));
197  return *this;
198  }
199  /// Conditionally append a pattern to tile the Op `opName` and fuse its
200  /// producers with tiling and fusion `options`.
201  CodegenStrategy &
202  tileAndFuseIf(bool b, StringRef opName, LinalgTilingAndFusionOptions options,
203  LinalgTransformationFilter::FilterFunction f = nullptr) {
204  return b ? tileAndFuse(opName, std::move(options), std::move(f)) : *this;
205  }
206  /// Append a pattern to add a level of tiling for Op `opName` with tiling

219  LinalgTransformationFilter::FilterFunction f = nullptr) {
220  return b ? tile(opName, std::move(options), std::move(f)) : *this;
221  }
222  /// Append a pattern to pad and hoist the operands of Op `opName` with padding
223  /// `options`.
224  CodegenStrategy &

251  LinalgTransformationFilter::FilterFunction f = nullptr) {

259  std::make_unique<Generalize>(opName, f));

297  bool vectorizePadding = false) {
