80 /// Create a linalg::GenericOp version of an n-D copy that can further tile,
81 /// lower to loops or vectorize, unlike the current implementation of
82 /// memref::CopyOp.
83 /// Do not depend on linalg::CopyOp that is getting deprecated.
84 static LogicalResult createLinalgCopyOp(OpBuilder &b, Location loc, Value from,
85  Value to) {
86  auto memrefTypeFrom = from.getType().cast<MemRefType>();
87  auto memrefTypeTo = to.getType().cast<MemRefType>();
88  if (!memrefTypeFrom || !memrefTypeTo ||
89  memrefTypeFrom.getRank() != memrefTypeTo.getRank())
90  return failure();
91  AffineMap id =
92  AffineMap::getMultiDimIdentityMap(memrefTypeTo.getRank(), b.getContext());
93  SmallVector<StringRef> iteratorTypes(memrefTypeTo.getRank(),
94  getParallelIteratorTypeName());
95  b.create<linalg::GenericOp>(loc,
96  /*inputs=*/from,
97  /*outputs=*/to,
98  /*indexingMaps=*/llvm::makeArrayRef({id, id}),
99  /*iteratorTypes=*/iteratorTypes,
100  [](OpBuilder &b, Location loc, ValueRange args) {
101  b.create<linalg::YieldOp>(loc, args.front());
102  });
103  return success();
104 }

106 void LinalgComprehensiveModuleBufferize::runOnOperation() {
107  auto options = std::make_unique<AnalysisBufferizationOptions>();
108  if (useAlloca) {
109  options->allocationFn = allocationFnUsingAlloca;
110  options->deallocationFn = [](OpBuilder &b, Location loc, Value v) {
111  return success();
112  };
113  }
114  // TODO: atm memref::CopyOp can be 200x slower than linalg::GenericOp.
115  // Once this perf bug is fixed more systematically, we can revisit.
116  if (useLinalgCopy)
117  options->memCpyFn = createLinalgCopyOp;
118 
119  options->allowReturnMemref = allowReturnMemref;
120  options->allowUnknownOps = allowUnknownOps;
121  options->analysisFuzzerSeed = analysisFuzzerSeed;
122  options->createDeallocs = createDeallocs;
123  options->fullyDynamicLayoutMaps = fullyDynamicLayoutMaps;
124  options->printConflicts = printConflicts;
125  options->testAnalysisOnly = testAnalysisOnly;
126 
127  // Enable InitTensorOp elimination.
128  if (initTensorElimination) {
129  options->addPostAnalysisStep<
130  linalg_ext::InsertSliceAnchoredInitTensorEliminationStep>();
131  }
132 
133  // Only certain scf.for ops are supported by the analysis.
134  options->addPostAnalysisStep<scf_ext::AssertScfForAliasingProperties>();
135 
136  ModuleOp moduleOp = getOperation();
137  applyEnablingTransformations(moduleOp);
138 
139  if (failed(runComprehensiveBufferize(moduleOp, std::move(options)))) {
140  signalPassFailure();
141  return;
142  }
143 
144  if (testAnalysisOnly)
145  return;
146 
147  OpPassManager cleanupPipeline("builtin.module");
148  cleanupPipeline.addPass(createCanonicalizerPass());
149  cleanupPipeline.addPass(createCSEPass());
150  cleanupPipeline.addPass(createLoopInvariantCodeMotionPass());
151  (void)runPipeline(cleanupPipeline, moduleOp);
152 }
