139 void IntegerPolyhedron::removeIdRange(unsigned idStart, unsigned idLimit) {
140  assert(idLimit < getNumCols() && "invalid id limit");
141 
142  if (idStart >= idLimit)
143  return;
144 
145  // We are going to be removing one or more identifiers from the range.
146  assert(idStart < numIds && "invalid idStart position");
147 
148  // TODO: Make 'removeIdRange' a lambda called from here.
149  // Remove eliminated identifiers from the constraints..
150  equalities.removeColumns(idStart, idLimit - idStart);
151  inequalities.removeColumns(idStart, idLimit - idStart);
152 
153  // Update members numDims, numSymbols and numIds.
154  unsigned numDimsEliminated = 0;
155  unsigned numLocalsEliminated = 0;
156  unsigned numColsEliminated = idLimit - idStart;
157  if (idStart < numDims) {
158  numDimsEliminated = std::min(numDims, idLimit) - idStart;
159  }
160  // Check how many local id's were removed. Note that our identifier order is
161  // [dims, symbols, locals]. Local id start at position numDims + numSymbols.
162  if (idLimit > numDims + numSymbols) {
163  numLocalsEliminated = std::min(
164  idLimit - std::max(idStart, numDims + numSymbols), getNumLocalIds());
165  }
166  unsigned numSymbolsEliminated =
167  numColsEliminated - numDimsEliminated - numLocalsEliminated;
168 
169  numDims -= numDimsEliminated;
170  numSymbols -= numSymbolsEliminated;
171  numIds = numIds - numColsEliminated;
172 }

232 /// Gather all lower and upper bounds of the identifier at `pos`, and
233 /// optionally any equalities on it. In addition, the bounds are to be
234 /// independent of identifiers in position range [`offset`, `offset` + `num`).
235 void IntegerPolyhedron::getLowerAndUpperBoundIndices(
236  unsigned pos, SmallVectorImpl<unsigned> *lbIndices,
237  SmallVectorImpl<unsigned> *ubIndices, SmallVectorImpl<unsigned> *eqIndices,
238  unsigned offset, unsigned num) const {
239  assert(pos < getNumIds() && "invalid position");
240  assert(offset + num < getNumCols() && "invalid range");
241 
242  // Checks for a constraint that has a non-zero coeff for the identifiers in
243  // the position range [offset, offset + num) while ignoring `pos`.
244  auto containsConstraintDependentOnRange = [&](unsigned r, bool isEq) {
245  unsigned c, f;
246  auto cst = isEq ? getEquality(r) : getInequality(r);
247  for (c = offset, f = offset + num; c < f; ++c) {
248  if (c == pos)
249  continue;
250  if (cst[c] != 0)
251  break;
252  }
253  return c < f;
254  };
255 
256  // Gather all lower bounds and upper bounds of the variable. Since the
257  // canonical form c_1*x_1 + c_2*x_2 + ... + c_0 >= 0, a constraint is a lower
258  // bound for x_i if c_i >= 1, and an upper bound if c_i <= -1.
259  for (unsigned r = 0, e = getNumInequalities(); r < e; r++) {
260  // The bounds are to be independent of [offset, offset + num) columns.
261  if (containsConstraintDependentOnRange(r, /*isEq=*/false))
262  continue;
263  if (atIneq(r, pos) >= 1) {
264  // Lower bound.
265  lbIndices->push_back(r);
266  } else if (atIneq(r, pos) <= -1) {
267  // Upper bound.
268  ubIndices->push_back(r);
269  }
270  }
271 
272  // An equality is both a lower and upper bound. Record any equalities
273  // involving the pos^th identifier.
274  if (!eqIndices)
275  return;
276 
277  for (unsigned r = 0, e = getNumEqualities(); r < e; r++) {
278  if (atEq(r, pos) == 0)
279  continue;
280  if (containsConstraintDependentOnRange(r, /*isEq=*/true))
281  continue;
282  eqIndices->push_back(r);
283  }
284 }

299 void IntegerPolyhedron::setAndEliminate(unsigned pos,
300  ArrayRef<int64_t> values) {
301  if (values.empty())
302  return;
303  assert(pos + values.size() <= getNumIds() &&
304  "invalid position or too many values");
305  // Setting x_j = p in sum_i a_i x_i + c is equivalent to adding p*a_j to the
306  // constant term and removing the id x_j. We do this for all the ids
307  // pos, pos + 1, ... pos + values.size() - 1.
308  for (unsigned r = 0, e = getNumInequalities(); r < e; r++)
309  for (unsigned i = 0, numVals = values.size(); i < numVals; ++i)
310  atIneq(r, getNumCols() - 1) += atIneq(r, pos + i) * values[i];
311  for (unsigned r = 0, e = getNumEqualities(); r < e; r++)
312  for (unsigned i = 0, numVals = values.size(); i < numVals; ++i)
313  atEq(r, getNumCols() - 1) += atEq(r, pos + i) * values[i];
314  removeIdRange(pos, pos + values.size());
315 }

339 // Normalizes the coefficient values across all columns in `rowIdx` by their
340 // GCD in equality or inequality constraints as specified by `isEq`.
341 template <bool isEq>
342 static void normalizeConstraintByGCD(IntegerPolyhedron *constraints,
343  unsigned rowIdx) {
344  auto at = [&](unsigned colIdx) -> int64_t {
345  return isEq ? constraints->atEq(rowIdx, colIdx)
346  : constraints->atIneq(rowIdx, colIdx);
347  };
348  uint64_t gcd = std::abs(at(0));
349  for (unsigned j = 1, e = constraints->getNumCols(); j < e; ++j) {
350  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(at(j)));
351  }
352  if (gcd > 0 && gcd != 1) {
353  for (unsigned j = 0, e = constraints->getNumCols(); j < e; ++j) {
354  int64_t v = at(j) / static_cast<int64_t>(gcd);
355  isEq ? constraints->atEq(rowIdx, j) = v
356  : constraints->atIneq(rowIdx, j) = v;
357  }
358  }
359 }

370 bool IntegerPolyhedron::hasInvalidConstraint() const {
371  assert(hasConsistentState());
372  auto check = [&](bool isEq) -> bool {
373  unsigned numCols = getNumCols();
374  unsigned numRows = isEq ? getNumEqualities() : getNumInequalities();
375  for (unsigned i = 0, e = numRows; i < e; ++i) {
376  unsigned j;
377  for (j = 0; j < numCols - 1; ++j) {
378  int64_t v = isEq ? atEq(i, j) : atIneq(i, j);
379  // Skip rows with non-zero variable coefficients.
380  if (v != 0)
381  break;
382  }
383  if (j < numCols - 1) {
384  continue;
385  }
386  // Check validity of constant term at 'numCols - 1' w.r.t 'isEq'.
387  // Example invalid constraints include: '1 == 0' or '-1 >= 0'
388  int64_t v = isEq ? atEq(i, numCols - 1) : atIneq(i, numCols - 1);
389  if ((isEq && v != 0) || (!isEq && v < 0)) {
390  return true;
391  }
392  }
393  return false;
394  };
395  if (check(/*isEq=*/true))
396  return true;
397  return check(/*isEq=*/false);
398 }

400 /// Eliminate identifier from constraint at `rowIdx` based on coefficient at
401 /// pivotRow, pivotCol. Columns in range [elimColStart, pivotCol) will not be
402 /// updated as they have already been eliminated.
403 static void eliminateFromConstraint(IntegerPolyhedron *constraints,
404  unsigned rowIdx, unsigned pivotRow,
405  unsigned pivotCol, unsigned elimColStart,
406  bool isEq) {
407  // Skip if equality 'rowIdx' if same as 'pivotRow'.
408  if (isEq && rowIdx == pivotRow)
409  return;
410  auto at = [&](unsigned i, unsigned j) -> int64_t {
411  return isEq ? constraints->atEq(i, j) : constraints->atIneq(i, j);
412  };
413  int64_t leadCoeff = at(rowIdx, pivotCol);
414  // Skip if leading coefficient at 'rowIdx' is already zero.
415  if (leadCoeff == 0)
416  return;
417  int64_t pivotCoeff = constraints->atEq(pivotRow, pivotCol);
418  int64_t sign = (leadCoeff * pivotCoeff > 0) ? -1 : 1;
419  int64_t lcm = mlir::lcm(pivotCoeff, leadCoeff);
420  int64_t pivotMultiplier = sign * (lcm / std::abs(pivotCoeff));
421  int64_t rowMultiplier = lcm / std::abs(leadCoeff);
422 
423  unsigned numCols = constraints->getNumCols();
424  for (unsigned j = 0; j < numCols; ++j) {
425  // Skip updating column 'j' if it was just eliminated.
426  if (j >= elimColStart && j < pivotCol)
427  continue;
428  int64_t v = pivotMultiplier * constraints->atEq(pivotRow, j) +
429  rowMultiplier * at(rowIdx, j);
430  isEq ? constraints->atEq(rowIdx, j) = v
431  : constraints->atIneq(rowIdx, j) = v;
432  }
433 }

435 /// Returns the position of the identifier that has the minimum <number of lower
436 /// bounds> times <number of upper bounds> from the specified range of
437 /// identifiers [start, end). It is often best to eliminate in the increasing
438 /// order of these counts when doing Fourier-Motzkin elimination since FM adds
439 /// that many new constraints.
440 static unsigned getBestIdToEliminate(const IntegerPolyhedron &cst,
441  unsigned start, unsigned end) {
442  assert(start < cst.getNumIds() && end < cst.getNumIds() + 1);
443 
444  auto getProductOfNumLowerUpperBounds = [&](unsigned pos) {
445  unsigned numLb = 0;
446  unsigned numUb = 0;
447  for (unsigned r = 0, e = cst.getNumInequalities(); r < e; r++) {
448  if (cst.atIneq(r, pos) > 0) {
449  ++numLb;
450  } else if (cst.atIneq(r, pos) < 0) {
451  ++numUb;
452  }
453  }
454  return numLb * numUb;
455  };
456 
457  unsigned minLoc = start;
458  unsigned min = getProductOfNumLowerUpperBounds(start);
459  for (unsigned c = start + 1; c < end; c++) {
460  unsigned numLbUbProduct = getProductOfNumLowerUpperBounds(c);
461  if (numLbUbProduct < min) {
462  min = numLbUbProduct;
463  minLoc = c;
464  }
465  }
466  return minLoc;
467 }

469 // Checks for emptiness of the set by eliminating identifiers successively and
470 // using the GCD test (on all equality constraints) and checking for trivially
471 // invalid constraints. Returns 'true' if the constraint system is found to be
472 // empty; false otherwise.
473 bool IntegerPolyhedron::isEmpty() const {
474  if (isEmptyByGCDTest() || hasInvalidConstraint())
475  return true;
476 
477  IntegerPolyhedron tmpCst(*this);
478 
479  // First, eliminate as many local variables as possible using equalities.
480  tmpCst.removeRedundantLocalVars();
481  if (tmpCst.isEmptyByGCDTest() || tmpCst.hasInvalidConstraint())
482  return true;
483 
484  // Eliminate as many identifiers as possible using Gaussian elimination.
485  unsigned currentPos = 0;
486  while (currentPos < tmpCst.getNumIds()) {
487  tmpCst.gaussianEliminateIds(currentPos, tmpCst.getNumIds());
488  ++currentPos;
489  // We check emptiness through trivial checks after eliminating each ID to
490  // detect emptiness early. Since the checks isEmptyByGCDTest() and
491  // hasInvalidConstraint() are linear time and single sweep on the constraint
492  // buffer, this appears reasonable - but can optimize in the future.
493  if (tmpCst.hasInvalidConstraint() || tmpCst.isEmptyByGCDTest())
494  return true;
495  }
496 
497  // Eliminate the remaining using FM.
498  for (unsigned i = 0, e = tmpCst.getNumIds(); i < e; i++) {
499  tmpCst.fourierMotzkinEliminate(
500  getBestIdToEliminate(tmpCst, 0, tmpCst.getNumIds()));
501  // Check for a constraint explosion. This rarely happens in practice, but
502  // this check exists as a safeguard against improperly constructed
503  // constraint systems or artificially created arbitrarily complex systems
504  // that aren't the intended use case for IntegerPolyhedron. This is
505  // needed since FM has a worst case exponential complexity in theory.
506  if (tmpCst.getNumConstraints() >= kExplosionFactor * getNumIds()) {
507  LLVM_DEBUG(llvm::dbgs() << "FM constraint explosion detected\n");
508  return false;
509  }
510 
511  // FM wouldn't have modified the equalities in any way. So no need to again
512  // run GCD test. Check for trivial invalid constraints.
513  if (tmpCst.hasInvalidConstraint())
514  return true;
515  }
516  return false;
517 }

519 // Runs the GCD test on all equality constraints. Returns 'true' if this test
520 // fails on any equality. Returns 'false' otherwise.
521 // This test can be used to disprove the existence of a solution. If it returns
522 // true, no integer solution to the equality constraints can exist.
523 //
524 // GCD test definition:
525 //
526 // The equality constraint:
527 //
528 // c_1*x_1 + c_2*x_2 + ... + c_n*x_n = c_0
529 //
530 // has an integer solution iff:
531 //
532 // GCD of c_1, c_2, ..., c_n divides c_0.
533 //
534 bool IntegerPolyhedron::isEmptyByGCDTest() const {
535  assert(hasConsistentState());
536  unsigned numCols = getNumCols();
537  for (unsigned i = 0, e = getNumEqualities(); i < e; ++i) {
538  uint64_t gcd = std::abs(atEq(i, 0));
539  for (unsigned j = 1; j < numCols - 1; ++j) {
540  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(atEq(i, j)));
541  }
542  int64_t v = std::abs(atEq(i, numCols - 1));
543  if (gcd > 0 && (v % gcd != 0)) {
544  return true;
545  }
546  }
547  return false;
548 }

550 // Returns a matrix where each row is a vector along which the polytope is
551 // bounded. The span of the returned vectors is guaranteed to contain all
552 // such vectors. The returned vectors are NOT guaranteed to be linearly
553 // independent. This function should not be called on empty sets.
554 //
555 // It is sufficient to check the perpendiculars of the constraints, as the set
556 // of perpendiculars which are bounded must span all bounded directions.
557 Matrix IntegerPolyhedron::getBoundedDirections() const {
558  // Note that it is necessary to add the equalities too (which the constructor
559  // does) even though we don't need to check if they are bounded; whether an
560  // inequality is bounded or not depends on what other constraints, including
561  // equalities, are present.
562  Simplex simplex(*this);
563 
564  assert(!simplex.isEmpty() && "It is not meaningful to ask whether a "
565  "direction is bounded in an empty set.");
566 
567  SmallVector<unsigned, 8> boundedIneqs;
568  // The constructor adds the inequalities to the simplex first, so this
569  // processes all the inequalities.
570  for (unsigned i = 0, e = getNumInequalities(); i < e; ++i) {
571  if (simplex.isBoundedAlongConstraint(i))
572  boundedIneqs.push_back(i);
573  }
574 
575  // The direction vector is given by the coefficients and does not include the
576  // constant term, so the matrix has one fewer column.
577  unsigned dirsNumCols = getNumCols() - 1;
578  Matrix dirs(boundedIneqs.size() + getNumEqualities(), dirsNumCols);
579 
580  // Copy the bounded inequalities.
581  unsigned row = 0;
582  for (unsigned i : boundedIneqs) {
583  for (unsigned col = 0; col < dirsNumCols; ++col)
584  dirs(row, col) = atIneq(i, col);
585  ++row;
586  }
587 
588  // Copy the equalities. All the equalities' perpendiculars are bounded.
589  for (unsigned i = 0, e = getNumEqualities(); i < e; ++i) {

629 /// Let this set be S. If S is bounded then we directly call into the GBR
630 /// sampling algorithm. Otherwise, there are some unbounded directions, i.e.,
631 /// vectors v such that S extends to infinity along v or -v. In this case we
632 /// use an algorithm described in the integer set library (isl) manual and used
633 /// by the isl_set_sample function in that library. The algorithm is:
634 ///
635 /// 1) Apply a unimodular transform T to S to obtain S*T, such that all
636 /// dimensions in which S*T is bounded lie in the linear span of a prefix of the
637 /// dimensions.
638 ///
639 /// 2) Construct a set B by removing all constraints that involve
640 /// the unbounded dimensions and then deleting the unbounded dimensions. Note
641 /// that B is a Bounded set.
642 ///
643 /// 3) Try to obtain a sample from B using the GBR sampling
644 /// algorithm. If no sample is found, return that S is empty.
645 ///
646 /// 4) Otherwise, substitute the obtained sample into S*T to obtain a set
647 /// C. C is a full-dimensional Cone and always contains a sample.
648 ///
649 /// 5) Obtain an integer sample from C.
650 ///
651 /// 6) Return T*v, where v is the concatenation of the samples from B and C.
652 ///
653 /// The following is a sketch of a proof that
654 /// a) If the algorithm returns empty, then S is empty.
655 /// b) If the algorithm returns a sample, it is a valid sample in S.
656 ///
657 /// The algorithm returns empty only if B is empty, in which case S*T is
658 /// certainly empty since B was obtained by removing constraints and then
659 /// deleting unconstrained dimensions from S*T. Since T is unimodular, a vector
660 /// v is in S*T iff T*v is in S. So in this case, since
661 /// S*T is empty, S is empty too.
662 ///
663 /// Otherwise, the algorithm substitutes the sample from B into S*T. All the
664 /// constraints of S*T that did not involve unbounded dimensions are satisfied
665 /// by this substitution. All dimensions in the linear span of the dimensions
666 /// outside the prefix are unbounded in S*T (step 1). Substituting values for
667 /// the bounded dimensions cannot make these dimensions bounded, and these are
668 /// the only remaining dimensions in C, so C is unbounded along every vector (in
669 /// the positive or negative direction, or both). C is hence a full-dimensional
670 /// cone and therefore always contains an integer point.
671 ///
672 /// Concatenating the samples from B and C gives a sample v in S*T, so the
673 /// returned sample T*v is a sample in S.
674 Optional<SmallVector<int64_t, 8>> IntegerPolyhedron::findIntegerSample() const {
675  // First, try the GCD test heuristic.
676  if (isEmptyByGCDTest())
677  return {};
678 
679  Simplex simplex(*this);
680  if (simplex.isEmpty())
681  return {};
682 
683  // For a bounded set, we directly call into the GBR sampling algorithm.
684  if (!simplex.isUnbounded())
685  return simplex.findIntegerSample();
686 
687  // The set is unbounded. We cannot directly use the GBR algorithm.
688  //
689  // m is a matrix containing, in each row, a vector in which S is
690  // bounded, such that the linear span of all these dimensions contains all
691  // bounded dimensions in S.
692  Matrix m = getBoundedDirections();
693  // In column echelon form, each row of m occupies only the first rank(m)
694  // columns and has zeros on the other columns. The transform T that brings S
695  // to column echelon form is unimodular as well, so this is a suitable
696  // transform to use in step 1 of the algorithm.
697  std::pair<unsigned, LinearTransform> result =
698  LinearTransform::makeTransformToColumnEchelon(std::move(m));
699  const LinearTransform &transform = result.second;
700  // 1) Apply T to S to obtain S*T.
701  IntegerPolyhedron transformedSet = transform.applyTo(*this);
702 
703  // 2) Remove the unbounded dimensions and constraints involving them to
704  // obtain a bounded set.
705  IntegerPolyhedron boundedSet(transformedSet);
706  unsigned numBoundedDims = result.first;
707  unsigned numUnboundedDims = getNumIds() - numBoundedDims;
708  removeConstraintsInvolvingSuffixDims(boundedSet, numUnboundedDims);
709  boundedSet.removeIdRange(numBoundedDims, boundedSet.getNumIds());
710 
711  // 3) Try to obtain a sample from the bounded set.
712  Optional<SmallVector<int64_t, 8>> boundedSample =
713  Simplex(boundedSet).findIntegerSample();
714  if (!boundedSample)
715  return {};
716  assert(boundedSet.containsPoint(*boundedSample) &&
717  "Simplex returned an invalid sample!");
718 
719  // 4) Substitute the values of the bounded dimensions into S*T to obtain a
720  // full-dimensional cone, which necessarily contains an integer sample.
721  transformedSet.setAndEliminate(0, *boundedSample);
722  IntegerPolyhedron &cone = transformedSet;
723 
724  // 5) Obtain an integer sample from the cone.
725  //
726  // We shrink the cone such that for any rational point in the shrunken cone,
727  // rounding up each of the point's coordinates produces a point that still
728  // lies in the original cone.
729  //
730  // Rounding up a point x adds a number e_i in [0, 1) to each coordinate x_i.
731  // For each inequality sum_i a_i x_i + c >= 0 in the original cone, the
732  // shrunken cone will have the inequality tightened by some amount s, such
733  // that if x satisfies the shrunken cone's tightened inequality, then x + e
734  // satisfies the original inequality, i.e.,
735  //
736  // sum_i a_i x_i + c + s >= 0 implies sum_i a_i (x_i + e_i) + c >= 0
737  //
738  // for any e_i values in [0, 1). In fact, we will handle the slightly more
739  // general case where e_i can be in [0, 1]. For example, consider the
740  // inequality 2x_1 - 3x_2 - 7x_3 - 6 >= 0, and let x = (3, 0, 0). How low
741  // could the LHS go if we added a number in [0, 1] to each coordinate? The LHS
742  // is minimized when we add 1 to the x_i with negative coefficient a_i and
743  // keep the other x_i the same. In the example, we would get x = (3, 1, 1),
744  // changing the value of the LHS by -3 + -7 = -10.
745  //
746  // In general, the value of the LHS can change by at most the sum of the
747  // negative a_i, so we accomodate this by shifting the inequality by this
748  // amount for the shrunken cone.
749  for (unsigned i = 0, e = cone.getNumInequalities(); i < e; ++i) {
750  for (unsigned j = 0; j < cone.getNumIds(); ++j) {
751  int64_t coeff = cone.atIneq(i, j);
752  if (coeff < 0)
753  cone.atIneq(i, cone.getNumIds()) += coeff;
754  }
755  }
756 
757  // Obtain an integer sample in the cone by rounding up a rational point from
758  // the shrunken cone. Shrinking the cone amounts to shifting its apex
759  // "inwards" without changing its "shape"; the shrunken cone is still a
760  // full-dimensional cone and is hence non-empty.
761  Simplex shrunkenConeSimplex(cone);
762  assert(!shrunkenConeSimplex.isEmpty() && "Shrunken cone cannot be empty!");
763 
764  // The sample will always exist since the shrunken cone is non-empty.
765  SmallVector<Fraction, 8> shrunkenConeSample =
766  *shrunkenConeSimplex.getRationalSample();
767 
768  SmallVector<int64_t, 8> coneSample(llvm::map_range(shrunkenConeSample, ceil));
769 
770  // 6) Return transform * concat(boundedSample, coneSample).
771  SmallVector<int64_t, 8> &sample = boundedSample.getValue();
772  sample.append(coneSample.begin(), coneSample.end());
773  return transform.postMultiplyWithColumn(sample);
774 }

857 /// Tightens inequalities given that we are dealing with integer spaces. This is
858 /// analogous to the GCD test but applied to inequalities. The constant term can
859 /// be reduced to the preceding multiple of the GCD of the coefficients, i.e.,
860 /// 64*i - 100 >= 0 => 64*i - 128 >= 0 (since 'i' is an integer). This is a
861 /// fast method - linear in the number of coefficients.
862 // Example on how this affects practical cases: consider the scenario:
863 // 64*i >= 100, j = 64*i; without a tightening, elimination of i would yield
864 // j >= 100 instead of the tighter (exact) j >= 128.
865 void IntegerPolyhedron::gcdTightenInequalities() {
866  unsigned numCols = getNumCols();
867  for (unsigned i = 0, e = getNumInequalities(); i < e; ++i) {
868  uint64_t gcd = std::abs(atIneq(i, 0));
869  for (unsigned j = 1; j < numCols - 1; ++j) {
870  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(atIneq(i, j)));
871  }
872  if (gcd > 0 && gcd != 1) {
873  int64_t gcdI = static_cast<int64_t>(gcd);
874  // Tighten the constant term and normalize the constraint by the GCD.
875  atIneq(i, numCols - 1) = mlir::floorDiv(atIneq(i, numCols - 1), gcdI);
876  for (unsigned j = 0, e = numCols - 1; j < e; ++j)
877  atIneq(i, j) /= gcdI;
878  }
879  }
880 }

883 // Returns the number of variables eliminated.
884 unsigned IntegerPolyhedron::gaussianEliminateIds(unsigned posStart,
885  unsigned posLimit) {

947 // A more complex check to eliminate redundant inequalities. Uses FourierMotzkin
948 // to check if a constraint is redundant.
949 void IntegerPolyhedron::removeRedundantInequalities() {
950  SmallVector<bool, 32> redun(getNumInequalities(), false);
951  // To check if an inequality is redundant, we replace the inequality by its
952  // complement (for eg., i - 1 >= 0 by i <= 0), and check if the resulting
953  // system is empty. If it is, the inequality is redundant.
954  IntegerPolyhedron tmpCst(*this);
955  for (unsigned r = 0, e = getNumInequalities(); r < e; r++) {
956  // Change the inequality to its complement.
957  negateInequality(&tmpCst, r);
958  tmpCst.atIneq(r, tmpCst.getNumCols() - 1)--;
959  if (tmpCst.isEmpty()) {
960  redun[r] = true;
961  // Zero fill the redundant inequality.
962  fillInequality(this, r, /*val=*/0);
963  fillInequality(&tmpCst, r, /*val=*/0);
964  } else {
965  // Reverse the change (to avoid recreating tmpCst each time).
966  tmpCst.atIneq(r, tmpCst.getNumCols() - 1)++;
967  negateInequality(&tmpCst, r);
968  }
969  }

1053 /// of the local ids in each set, without changing the set of points that
1054 /// lie in `this` and `other`.
1055 ///
1056 /// To detect local ids that always take the same in both sets, each local id is
1057 /// represented as a floordiv with constant denominator in terms of other ids.
1058 /// After extracting these divisions, local ids with the same division
1059 /// representation are considered duplicate and are merged. It is possible that
1060 /// division representation for some local id cannot be obtained, and thus these
1061 /// local ids are not considered for detecting duplicates.
1062 void IntegerPolyhedron::mergeLocalIds(IntegerPolyhedron &other) {
1063  assert(getNumDimIds() == other.getNumDimIds() &&
1064  "Number of dimension ids should match");
1065  assert(getNumSymbolIds() == other.getNumSymbolIds() &&
1066  "Number of symbol ids should match");
1067 
1068  IntegerPolyhedron &polyA = *this;
1069  IntegerPolyhedron &polyB = other;
1070 
1071  // Merge local ids of polyA and polyB without using division information,
1072  // i.e. append local ids of `polyB` to `polyA` and insert local ids of `polyA`
1073  // to `polyB` at start of its local ids.
1074  unsigned initLocals = polyA.getNumLocalIds();
1075  insertLocalId(polyA.getNumLocalIds(), polyB.getNumLocalIds());
1076  polyB.insertLocalId(0, initLocals);
1077 
1078  // Get division representations from each poly.
1079  std::vector<SmallVector<int64_t, 8>> divsA, divsB;
1080  SmallVector<unsigned, 4> denomsA, denomsB;
1081  polyA.getLocalReprs(divsA, denomsA);
1082  polyB.getLocalReprs(divsB, denomsB);
1083 
1084  // Copy division information for polyB into `divsA` and `denomsA`, so that
1085  // these have the combined division information of both polys. Since newly
1086  // added local variables in polyA and polyB have no constraints, they will not
1087  // have any division representation.
1088  std::copy(divsB.begin() + initLocals, divsB.end(),
1089  divsA.begin() + initLocals);
1090  std::copy(denomsB.begin() + initLocals, denomsB.end(),
1091  denomsA.begin() + initLocals);
1092

1094  // them as the same identifier.
1095  auto merge = [&polyA, &polyB](unsigned i, unsigned j) -> bool {
1096  eliminateRedundantLocalId(polyA, i, j);
1097  eliminateRedundantLocalId(polyB, i, j);
1098  return true;
1099  };
1100 
1101  // Merge all divisions by removing duplicate divisions.
1102  unsigned localOffset = getIdKindOffset(IdKind::Local);
1103  removeDuplicateDivs(divsA, denomsA, localOffset, merge);
1104 }

1610 /// Eliminates identifier at the specified position using Fourier-Motzkin
1611 /// variable elimination. This technique is exact for rational spaces but
1612 /// conservative (in "rare" cases) for integer spaces. The operation corresponds
1613 /// to a projection operation yielding the (convex) set of integer points
1614 /// contained in the rational shadow of the set. An emptiness test that relies
1615 /// on this method will guarantee emptiness, i.e., it disproves the existence of
1616 /// a solution if it says it's empty.
1617 /// If a non-null isResultIntegerExact is passed, it is set to true if the
1618 /// result is also integer exact. If it's set to false, the obtained solution
1619 /// *may* not be exact, i.e., it may contain integer points that do not have an
1620 /// integer pre-image in the original set.
1621 ///
1622 /// Eg:
1623 /// j >= 0, j <= i + 1
1624 /// i >= 0, i <= N + 1
1625 /// Eliminating i yields,
1626 /// j >= 0, 0 <= N + 1, j - 1 <= N + 1
1627 ///
1628 /// If darkShadow = true, this method computes the dark shadow on elimination;
1629 /// the dark shadow is a convex integer subset of the exact integer shadow. A
1630 /// non-empty dark shadow proves the existence of an integer solution. The
1631 /// elimination in such a case could however be an under-approximation, and thus
1632 /// should not be used for scanning sets or used by itself for dependence
1633 /// checking.
1634 ///
1635 /// Eg: 2-d set, * represents grid points, 'o' represents a point in the set.
1636 /// ^
1637 /// |
1638 /// | * * * * o o
1639 /// i | * * o o o o
1640 /// | o * * * * *
1641 /// --------------->
1642 /// j ->
1643 ///
1644 /// Eliminating i from this system (projecting on the j dimension):
1645 /// rational shadow / integer light shadow: 1 <= j <= 6
1646 /// dark shadow: 3 <= j <= 6
1647 /// exact integer shadow: j = 1 \union 3 <= j <= 6
1648 /// holes/splinters: j = 2
1649 ///
1650 /// darkShadow = false, isResultIntegerExact = nullptr are default values.

1727  unsigned lcmProducts = 1;

1730  // For each lower bound, lb <= c_l*x, and each upper bound c_u*x <= ub, (note
1731  // that c_l, c_u >= 1) we have:
1732  // lb*lcm(c_l, c_u)/c_l <= lcm(c_l, c_u)*x <= ub*lcm(c_l, c_u)/c_u
1733  // We thus generate a constraint:
1734  // lcm(c_l, c_u)/c_l*lb <= lcm(c_l, c_u)/c_u*ub.

1778  for (unsigned l = 0, e = getNumCols(); l < e; l++) {

1780  continue;
1781  ineq.push_back(atIneq(nbPos, l));
1782  }
1783  newPoly.addInequality(ineq);
1784  }
1785

1787  lbIndices.size() * ubIndices.size() + nbIndices.size());

1790  for (unsigned r = 0, e = getNumEqualities(); r < e; r++) {

1792  eq.reserve(newPoly.getNumCols());

1795  continue;
1796  eq.push_back(atEq(r, l));
1797  }
1798  newPoly.addEquality(eq);

1896 // Computes the bounding box with respect to 'other' by finding the min of the
1897 // lower bounds and the max of the upper bounds along each of the dimensions.
1898 LogicalResult
1899 IntegerPolyhedron::unionBoundingBox(const IntegerPolyhedron &otherCst) {
1900  assert(otherCst.getNumDimIds() == numDims && "dims mismatch");
1901  assert(otherCst.getNumLocalIds() == 0 && "local ids not supported here");
1902  assert(getNumLocalIds() == 0 && "local ids not supported yet here");
1903 
1904  // Get the constraints common to both systems; these will be added as is to
1905  // the union.
1906  IntegerPolyhedron commonCst;
1907  getCommonConstraints(*this, otherCst, commonCst);
1908 
1909  std::vector<SmallVector<int64_t, 8>> boundingLbs;
1910  std::vector<SmallVector<int64_t, 8>> boundingUbs;
1911  boundingLbs.reserve(2 * getNumDimIds());
1912  boundingUbs.reserve(2 * getNumDimIds());
1913 
1914  // To hold lower and upper bounds for each dimension.
1915  SmallVector<int64_t, 4> lb, otherLb, ub, otherUb;
1916  // To compute min of lower bounds and max of upper bounds for each dimension.
1917  SmallVector<int64_t, 4> minLb(getNumSymbolIds() + 1);
1918  SmallVector<int64_t, 4> maxUb(getNumSymbolIds() + 1);
1919  // To compute final new lower and upper bounds for the union.
1920  SmallVector<int64_t, 8> newLb(getNumCols()), newUb(getNumCols());
1921 
1922  int64_t lbFloorDivisor, otherLbFloorDivisor;
1923  for (unsigned d = 0, e = getNumDimIds(); d < e; ++d) {
1924  auto extent = getConstantBoundOnDimSize(d, &lb, &lbFloorDivisor, &ub);
1925  if (!extent.hasValue())
1926  // TODO: symbolic extents when necessary.
1927  // TODO: handle union if a dimension is unbounded.
1928  return failure();
1929 
1930  auto otherExtent = otherCst.getConstantBoundOnDimSize(
1931  d, &otherLb, &otherLbFloorDivisor, &otherUb);
1932  if (!otherExtent.hasValue() || lbFloorDivisor != otherLbFloorDivisor)
1933  // TODO: symbolic extents when necessary.
1934  return failure();
1935 
1936  assert(lbFloorDivisor > 0 && "divisor always expected to be positive");
1937 
1938  auto res = compareBounds(lb, otherLb);
1939  // Identify min.
1940  if (res == BoundCmpResult::Less || res == BoundCmpResult::Equal) {
1941  minLb = lb;
1942  // Since the divisor is for a floordiv, we need to convert to ceildiv,
1943  // i.e., i >= expr floordiv div <=> i >= (expr - div + 1) ceildiv div <=>
1944  // div * i >= expr - div + 1.
1945  minLb.back() -= lbFloorDivisor - 1;
1946  } else if (res == BoundCmpResult::Greater) {
1947  minLb = otherLb;
1948  minLb.back() -= otherLbFloorDivisor - 1;
1949  } else {
1950  // Uncomparable - check for constant lower/upper bounds.
1951  auto constLb = getConstantBound(BoundType::LB, d);
1952  auto constOtherLb = otherCst.getConstantBound(BoundType::LB, d);
1953  if (!constLb.hasValue() || !constOtherLb.hasValue())
1954  return failure();
1955  std::fill(minLb.begin(), minLb.end(), 0);
1956  minLb.back() = std::min(constLb.getValue(), constOtherLb.getValue());
1957  }
1958 
1959  // Do the same for ub's but max of upper bounds. Identify max.
1960  auto uRes = compareBounds(ub, otherUb);
1961  if (uRes == BoundCmpResult::Greater || uRes == BoundCmpResult::Equal) {
1962  maxUb = ub;
1963  } else if (uRes == BoundCmpResult::Less) {
1964  maxUb = otherUb;
1965  } else {
1966  // Uncomparable - check for constant lower/upper bounds.
1967  auto constUb = getConstantBound(BoundType::UB, d);
1968  auto constOtherUb = otherCst.getConstantBound(BoundType::UB, d);
1969  if (!constUb.hasValue() || !constOtherUb.hasValue())
1970  return failure();
1971  std::fill(maxUb.begin(), maxUb.end(), 0);
1972  maxUb.back() = std::max(constUb.getValue(), constOtherUb.getValue());
1973  }
1974 
1975  std::fill(newLb.begin(), newLb.end(), 0);
1976  std::fill(newUb.begin(), newUb.end(), 0);
1977 
1978  // The divisor for lb, ub, otherLb, otherUb at this point is lbDivisor,
1979  // and so it's the divisor for newLb and newUb as well.
1980  newLb[d] = lbFloorDivisor;
1981  newUb[d] = -lbFloorDivisor;
1982  // Copy over the symbolic part + constant term.
1983  std::copy(minLb.begin(), minLb.end(), newLb.begin() + getNumDimIds());
1984  std::transform(newLb.begin() + getNumDimIds(), newLb.end(),
1985  newLb.begin() + getNumDimIds(), std::negate<int64_t>());
1986  std::copy(maxUb.begin(), maxUb.end(), newUb.begin() + getNumDimIds());
1987 
1988  boundingLbs.push_back(newLb);
1989  boundingUbs.push_back(newUb);
1990  }
1991 
1992  // Clear all constraints and add the lower/upper bounds for the bounding box.
1993  clearConstraints();
1994  for (unsigned d = 0, e = getNumDimIds(); d < e; ++d) {
1995  addInequality(boundingLbs[d]);
1996  addInequality(boundingUbs[d]);
1997  }
1998 
1999  // Add the constraints that were common to both systems.
2000  append(commonCst);
2001  removeTrivialRedundancy();
2002 
2003  // TODO: copy over pure symbolic constraints from this and 'other' over to the
2004  // union (since the above are just the union along dimensions); we shouldn't
2005  // be discarding any other constraints on the symbols.
2006 
2007  return success();
2008 }
