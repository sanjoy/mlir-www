63 /// This class represents a scope for named AST decls. A scope determines the
64 /// visibility and lifetime of a named declaration.
65 class DeclScope {
66 public:
67  /// Create a new scope with an optional parent scope.
68  DeclScope(DeclScope *parent = nullptr) : parent(parent) {}
69 
70  /// Return the parent scope of this scope, or nullptr if there is no parent.
71  DeclScope *getParentScope() { return parent; }
72  const DeclScope *getParentScope() const { return parent; }
73 
74  /// Return all of the decls within this scope.
75  auto getDecls() const { return llvm::make_second_range(decls); }
76 
77  /// Add a new decl to the scope.
78  void add(Decl *decl);
79 
80  /// Lookup a decl with the given name starting from this scope. Returns
81  /// nullptr if no decl could be found.
82  Decl *lookup(StringRef name);
83  template <typename T> T *lookup(StringRef name) {
84  return dyn_cast_or_null<T>(lookup(name));
85  }
86  const Decl *lookup(StringRef name) const {
87  return const_cast<DeclScope *>(this)->lookup(name);
88  }
89  template <typename T> const T *lookup(StringRef name) const {
90  return dyn_cast_or_null<T>(lookup(name));
91  }
92 
93 private:
94  /// The parent scope, or null if this is a top-level scope.
95  DeclScope *parent;
96  /// The decls defined within this scope.
97  llvm::StringMap<Decl *> decls;
98 };

104 /// This class represents a base AST node. All AST nodes are derived from this
105 /// class, and it contains many of the base functionality for interacting with
106 /// nodes.
107 class Node {
108 public:
109  /// This CRTP class provides several utilies when defining new AST nodes.
110  template <typename T, typename BaseT> class NodeBase : public BaseT {
111  public:
112  using Base = NodeBase<T, BaseT>;
113 
114  /// Provide type casting support.
115  static bool classof(const Node *node) {
116  return node->getTypeID() == TypeID::get<T>();
117  }
118 
119  protected:
120  template <typename... Args>
121  explicit NodeBase(llvm::SMRange loc, Args &&...args)
122  : BaseT(TypeID::get<T>(), loc, std::forward<Args>(args)...) {}
123  };
124 
125  /// Return the type identifier of this node.
126  TypeID getTypeID() const { return typeID; }
127 
128  /// Return the location of this node.
129  llvm::SMRange getLoc() const { return loc; }
130 
131  /// Print this node to the given stream.
132  void print(raw_ostream &os) const;
133 
134 protected:
135  Node(TypeID typeID, llvm::SMRange loc) : typeID(typeID), loc(loc) {}
136 
137 private:
138  /// A unique type identifier for this node.
139  TypeID typeID;
140 
141  /// The location of this node.
142  llvm::SMRange loc;
143 };

426 /// This expression represents the structural form of an MLIR Operation. It
427 /// represents either an input operation to match, or an operation to create
428 /// within a rewrite.
429 class OperationExpr final
430  : public Node::NodeBase<OperationExpr, Expr>,
431  private llvm::TrailingObjects<OperationExpr, Expr *,
432  NamedAttributeDecl *> {
433 public:
434  static OperationExpr *create(Context &ctx, llvm::SMRange loc,
435  const OpNameDecl *nameDecl,
436  ArrayRef<Expr *> operands,
437  ArrayRef<Expr *> resultTypes,
438  ArrayRef<NamedAttributeDecl *> attributes);
439 
440  /// Return the name of the operation, or None if there isn't one.
441  Optional<StringRef> getName() const;
442 
443  /// Return the declaration of the operation name.
444  const OpNameDecl *getNameDecl() const { return nameDecl; }
445 
446  /// Return the location of the name of the operation expression, or an invalid
447  /// location if there isn't a name.
448  llvm::SMRange getNameLoc() const { return nameLoc; }
449 
450  /// Return the operands of this operation.
451  MutableArrayRef<Expr *> getOperands() {
452  return {getTrailingObjects<Expr *>(), numOperands};
453  }
454  ArrayRef<Expr *> getOperands() const {
455  return const_cast<OperationExpr *>(this)->getOperands();
456  }
457 
458  /// Return the result types of this operation.
459  MutableArrayRef<Expr *> getResultTypes() {
460  return {getTrailingObjects<Expr *>() + numOperands, numResultTypes};
461  }
462  MutableArrayRef<Expr *> getResultTypes() const {
463  return const_cast<OperationExpr *>(this)->getResultTypes();
464  }
465 
466  /// Return the attributes of this operation.
467  MutableArrayRef<NamedAttributeDecl *> getAttributes() {
468  return {getTrailingObjects<NamedAttributeDecl *>(), numAttributes};
469  }
470  MutableArrayRef<NamedAttributeDecl *> getAttributes() const {
471  return const_cast<OperationExpr *>(this)->getAttributes();
472  }
473 
474 private:
475  OperationExpr(llvm::SMRange loc, Type type, const OpNameDecl *nameDecl,
476  unsigned numOperands, unsigned numResultTypes,
477  unsigned numAttributes, llvm::SMRange nameLoc)
478  : Base(loc, type), nameDecl(nameDecl), numOperands(numOperands),
479  numResultTypes(numResultTypes), numAttributes(numAttributes),
480  nameLoc(nameLoc) {}
481 
482  /// The name decl of this expression.
483  const OpNameDecl *nameDecl;
484 
485  /// The number of operands, result types, and attributes of the operation.
486  unsigned numOperands, numResultTypes, numAttributes;
487 
488  /// The location of the operation name in the expression if it has a name.
489  llvm::SMRange nameLoc;
490 
491  /// TrailingObject utilities.
492  friend llvm::TrailingObjects<OperationExpr, Expr *, NamedAttributeDecl *>;
493  size_t numTrailingObjects(OverloadToken<Expr *>) const {
494  return numOperands + numResultTypes;
495  }
496 };
