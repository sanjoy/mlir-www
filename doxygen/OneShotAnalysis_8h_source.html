23 /// PostAnalysisSteps can be registered with `BufferizationOptions` and are
24 /// executed after the analysis, but before bufferization. They can be used to
25 /// implement custom dialect-specific optimizations.
26 struct PostAnalysisStep {
27  virtual ~PostAnalysisStep() = default;
28 
29  /// Run the post analysis step. This function may modify the IR, but must keep
30  /// `aliasInfo` consistent. Newly created operations and operations that
31  /// should be re-analyzed must be added to `newOps`.
32  virtual LogicalResult run(Operation *op, BufferizationState &state,
33  BufferizationAliasInfo &aliasInfo,
34  SmallVector<Operation *> &newOps) = 0;
35 };

58 /// The BufferizationAliasInfo class maintains a list of buffer aliases and
59 /// equivalence classes to support bufferization.
60 class BufferizationAliasInfo {
61 public:
62  explicit BufferizationAliasInfo(Operation *rootOp);
63 
64  // BufferizationAliasInfo should be passed as a reference.
65  BufferizationAliasInfo(const BufferizationAliasInfo &) = delete;
66 
67  /// Add a new entry for `v` in the `aliasInfo` and `equivalentInfo`. In the
68  /// beginning the alias and equivalence sets only contain `v` itself.
69  void createAliasInfoEntry(Value v);
70 
71  /// Insert an info entry for `newValue` and merge its alias set with that of
72  /// `alias`.
73  void insertNewBufferAlias(Value newValue, Value alias);
74 
75  /// Insert an info entry for `newValue` and merge its alias set with that of
76  /// `alias`. Additionally, merge their equivalence classes.
77  void insertNewBufferEquivalence(Value newValue, Value alias);
78 
79  /// Set the inPlace bufferization spec to true.
80  /// Merge result's and operand's aliasing sets and iterate to a fixed point.
81  void bufferizeInPlace(OpOperand &operand, BufferizationState &state);
82 
83  /// Set the inPlace bufferization spec to false.
84  void bufferizeOutOfPlace(OpOperand &operand);
85 
86  /// Return true if `v1` and `v2` bufferize to equivalent buffers.
87  bool areEquivalentBufferizedValues(Value v1, Value v2) const {
88  return equivalentInfo.isEquivalent(v1, v2);
89  }
90 
91  /// Union the alias sets of `v1` and `v2`.
92  void unionAliasSets(Value v1, Value v2) { aliasInfo.unionSets(v1, v2); }
93 
94  /// Union the equivalence classes of `v1` and `v2`.
95  void unionEquivalenceClasses(Value v1, Value v2) {
96  equivalentInfo.unionSets(v1, v2);
97  }
98 
99  /// Apply `fun` to all the members of the equivalence class of `v`.
100  void applyOnEquivalenceClass(Value v, function_ref<void(Value)> fun) const;
101 
102  /// Apply `fun` to all aliases of `v`.
103  void applyOnAliases(Value v, function_ref<void(Value)> fun) const;
104 
105  /// Mark a value as in-place bufferized.
106  void markInPlace(OpOperand &o) { inplaceBufferized.insert(&o); }
107 
108  /// Return `true` if a value was marked as in-place bufferized.
109  bool isInPlace(OpOperand &opOperand) const;
110 
111 private:
112  /// llvm::EquivalenceClasses wants comparable elements. This comparator uses
113  /// uses pointer comparison on the defining op. This is a poor man's
114  /// comparison but it's not like UnionFind needs ordering anyway.
115  struct ValueComparator {
116  bool operator()(const Value &lhs, const Value &rhs) const {
117  return lhs.getImpl() < rhs.getImpl();
118  }
119  };
120 
121  using EquivalenceClassRangeType = llvm::iterator_range<
122  llvm::EquivalenceClasses<Value, ValueComparator>::member_iterator>;
123  /// Check that aliasInfo for `v` exists and return a reference to it.
124  EquivalenceClassRangeType getAliases(Value v) const;
125 
126  /// Set of all OpResults that were decided to bufferize in-place.
127  llvm::DenseSet<OpOperand *> inplaceBufferized;
128 
129  /// Auxiliary structure to store all the values a given value may alias with.
130  /// Alias information is "may be" conservative: In the presence of branches, a
131  /// value may alias with one of multiple other values. The concrete aliasing
132  /// value may not even be known at compile time. All such values are
133  /// considered to be aliases.
134  llvm::EquivalenceClasses<Value, ValueComparator> aliasInfo;
135 
136  /// Auxiliary structure to store all the equivalent buffer classes. Equivalent
137  /// buffer information is "must be" conservative: Only if two values are
138  /// guaranteed to be equivalent at runtime, they said to be equivalent. It is
139  /// possible that, in the presence of branches, it cannot be determined
140  /// statically if two values are equivalent. In that case, the values are
141  /// considered to be not equivalent.
142  llvm::EquivalenceClasses<Value, ValueComparator> equivalentInfo;
143 };
