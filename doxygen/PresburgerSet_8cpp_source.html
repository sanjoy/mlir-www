106 // Return the intersection of this set with the given set.
107 //
108 // We directly compute (S_1 or S_2 ...) and (T_1 or T_2 ...)
109 // as (S_1 and T_1) or (S_1 and T_2) or ...
110 //
111 // If S_i or T_j have local variables, then S_i and T_j contains the local
112 // variables of both.
113 PresburgerSet PresburgerSet::intersect(const PresburgerSet &set) const {
114  assertDimensionsCompatible(set, *this);
115 
116  PresburgerSet result(getNumDimIds(), getNumSymbolIds());
117  for (const IntegerPolyhedron &csA : integerPolyhedrons) {
118  for (const IntegerPolyhedron &csB : set.integerPolyhedrons) {
119  IntegerPolyhedron csACopy = csA, csBCopy = csB;
120  csACopy.mergeLocalIds(csBCopy);
121  csACopy.append(csBCopy);
122  if (!csACopy.isEmpty())
123  result.unionPolyInPlace(csACopy);
124  }
125  }
126  return result;
127 }

139 ///
140 /// The complement of a_1 x_1 + ... + a_n x_ + c >= 0 is
141 /// a_1 x_1 + ... + a_n x_ + c < 0, i.e., -a_1 x_1 - ... - a_n x_ - c - 1 >= 0,
142 /// since all the variables are constrained to be integers.

144  SmallVector<int64_t, 8> coeffs;

152 /// Return the set difference b \ s and accumulate the result into `result`.
153 /// `simplex` must correspond to b.
154 ///
155 /// In the following, U denotes union, ^ denotes intersection, \ denotes set
156 /// difference and ~ denotes complement.
157 /// Let b be the IntegerPolyhedron and s = (U_i s_i) be the set. We want
158 /// b \ (U_i s_i).
159 ///
160 /// Let s_i = ^_j s_ij, where each s_ij is a single inequality. To compute
161 /// b \ s_i = b ^ ~s_i, we partition s_i based on the first violated inequality:
162 /// ~s_i = (~s_i1) U (s_i1 ^ ~s_i2) U (s_i1 ^ s_i2 ^ ~s_i3) U ...
163 /// And the required result is (b ^ ~s_i1) U (b ^ s_i1 ^ ~s_i2) U ...
164 /// We recurse by subtracting U_{j > i} S_j from each of these parts and
165 /// returning the union of the results. Each equality is handled as a
166 /// conjunction of two inequalities.
167 ///
168 /// Note that the same approach works even if an inequality involves a floor
169 /// division. For example, the complement of x <= 7*floor(x/7) is still
170 /// x > 7*floor(x/7). Since b \ s_i contains the inequalities of both b and s_i
171 /// (or the complements of those inequalities), b \ s_i may contain the
172 /// divisions present in both b and s_i. Therefore, we need to add the local
173 /// division variables of both b and s_i to each part in the result. This means
174 /// adding the local variables of both b and s_i, as well as the corresponding
175 /// division inequalities to each part. Since the division inequalities are
176 /// added to each part, we can skip the parts where the complement of any
177 /// division inequality is added, as these parts will become empty anyway.
178 ///
179 /// As a heuristic, we try adding all the constraints and check if simplex
180 /// says that the intersection is empty. If it is, then subtracting this Poly is
181 /// a no-op and we just skip it. Also, in the process we find out that some
182 /// constraints are redundant. These redundant constraints are ignored.
183 ///
184 /// b and simplex are callee saved, i.e., their values on return are
185 /// semantically equivalent to their values when the function is called.
186 static void subtractRecursively(IntegerPolyhedron &b, Simplex &simplex,
187  const PresburgerSet &s, unsigned i,
188  PresburgerSet &result) {
189  if (i == s.getNumPolys()) {
190  result.unionPolyInPlace(b);
191  return;
192  }
193  IntegerPolyhedron sI = s.getIntegerPolyhedron(i);
194 
195  // Below, we append some additional constraints and ids to b. We want to
196  // rollback b to its initial state before returning, which we will do by
197  // removing all constraints beyond the original number of inequalities
198  // and equalities, so we store these counts first.
199  const unsigned bInitNumIneqs = b.getNumInequalities();
200  const unsigned bInitNumEqs = b.getNumEqualities();
201  const unsigned bInitNumLocals = b.getNumLocalIds();
202  // Similarly, we also want to rollback simplex to its original state.
203  const unsigned initialSnapshot = simplex.getSnapshot();
204 
205  // Automatically restore the original state when we return.
206  auto restoreState = [&]() {
207  b.removeIdRange(IntegerPolyhedron::IdKind::Local, bInitNumLocals,
208  b.getNumLocalIds());
209  b.removeInequalityRange(bInitNumIneqs, b.getNumInequalities());
210  b.removeEqualityRange(bInitNumEqs, b.getNumEqualities());
211  simplex.rollback(initialSnapshot);
212  };
213 
214  // Find out which inequalities of sI correspond to division inequalities for
215  // the local variables of sI.
216  std::vector<MaybeLocalRepr> repr(sI.getNumLocalIds());
217  sI.getLocalReprs(repr);
218 
219  // Add sI's locals to b, after b's locals. Also add b's locals to sI, before
220  // sI's locals.
221  b.mergeLocalIds(sI);
222 
223  // Mark which inequalities of sI are division inequalities and add all such
224  // inequalities to b.
225  llvm::SmallBitVector isDivInequality(sI.getNumInequalities());
226  for (MaybeLocalRepr &maybeInequality : repr) {
227  assert(maybeInequality.kind == ReprKind::Inequality &&
228  "Subtraction is not supported when a representation of the local "
229  "variables of the subtrahend cannot be found!");
230  auto lb = maybeInequality.repr.inEqualityPair.lowerBoundIdx;
231  auto ub = maybeInequality.repr.inEqualityPair.upperBoundIdx;
232 
233  b.addInequality(sI.getInequality(lb));
234  b.addInequality(sI.getInequality(ub));
235 
236  assert(lb != ub &&
237  "Upper and lower bounds must be different inequalities!");
238  isDivInequality[lb] = true;
239  isDivInequality[ub] = true;
240  }
241 
242  unsigned offset = simplex.getNumConstraints();
243  unsigned numLocalsAdded = b.getNumLocalIds() - bInitNumLocals;
244  simplex.appendVariable(numLocalsAdded);
245 
246  unsigned snapshotBeforeIntersect = simplex.getSnapshot();
247  simplex.intersectIntegerPolyhedron(sI);
248 
249  if (simplex.isEmpty()) {
250  /// b ^ s_i is empty, so b \ s_i = b. We move directly to i + 1.
251  /// We are ignoring level i completely, so we restore the state
252  /// *before* going to level i + 1.
253  restoreState();
254  subtractRecursively(b, simplex, s, i + 1, result);
255  return;
256  }
257 
258  simplex.detectRedundant();
259 
260  // Equalities are added to simplex as a pair of inequalities.
261  unsigned totalNewSimplexInequalities =
262  2 * sI.getNumEqualities() + sI.getNumInequalities();
263  llvm::SmallBitVector isMarkedRedundant(totalNewSimplexInequalities);
264  for (unsigned j = 0; j < totalNewSimplexInequalities; j++)
265  isMarkedRedundant[j] = simplex.isMarkedRedundant(offset + j);
266 
267  simplex.rollback(snapshotBeforeIntersect);
268 
269  // Recurse with the part b ^ ~ineq. Note that b is modified throughout
270  // subtractRecursively. At the time this function is called, the current b is
271  // actually equal to b ^ s_i1 ^ s_i2 ^ ... ^ s_ij, and ineq is the next
272  // inequality, s_{i,j+1}. This function recurses into the next level i + 1
273  // with the part b ^ s_i1 ^ s_i2 ^ ... ^ s_ij ^ ~s_{i,j+1}.
274  auto recurseWithInequality = [&, i](ArrayRef<int64_t> ineq) {
275  size_t snapshot = simplex.getSnapshot();
276  b.addInequality(ineq);
277  simplex.addInequality(ineq);
278  subtractRecursively(b, simplex, s, i + 1, result);
279  b.removeInequality(b.getNumInequalities() - 1);
280  simplex.rollback(snapshot);
281  };
282 
283  // For each inequality ineq, we first recurse with the part where ineq
284  // is not satisfied, and then add the ineq to b and simplex because
285  // ineq must be satisfied by all later parts.
286  auto processInequality = [&](ArrayRef<int64_t> ineq) {
287  recurseWithInequality(getComplementIneq(ineq));
288  b.addInequality(ineq);
289  simplex.addInequality(ineq);
290  };
291 
292  // Process all the inequalities, ignoring redundant inequalities and division
293  // inequalities. The result is correct whether or not we ignore these, but
294  // ignoring them makes the result simpler.
295  for (unsigned j = 0, e = sI.getNumInequalities(); j < e; j++) {
296  if (isMarkedRedundant[j])
297  continue;
298  if (isDivInequality[j])
299  continue;
300  processInequality(sI.getInequality(j));
301  }
302 
303  offset = sI.getNumInequalities();
304  for (unsigned j = 0, e = sI.getNumEqualities(); j < e; ++j) {
305  ArrayRef<int64_t> coeffs = sI.getEquality(j);
306  // For each equality, process the positive and negative inequalities that
307  // make up this equality. If Simplex found an inequality to be redundant, we
308  // skip it as above to make the result simpler. Divisions are always
309  // represented in terms of inequalities and not equalities, so we do not
310  // check for division inequalities here.
311  if (!isMarkedRedundant[offset + 2 * j])
312  processInequality(coeffs);
313  if (!isMarkedRedundant[offset + 2 * j + 1])
314  processInequality(getNegatedCoeffs(coeffs));
315  }
316 
317  restoreState();
318 }

390 PresburgerSet PresburgerSet::coalesce() const {
391  PresburgerSet newSet =
392  PresburgerSet::getEmptySet(getNumDimIds(), getNumSymbolIds());
393  llvm::SmallBitVector isRedundant(getNumPolys());
394 
395  for (unsigned i = 0, e = integerPolyhedrons.size(); i < e; ++i) {
396  if (isRedundant[i])
397  continue;
398  Simplex simplex(integerPolyhedrons[i]);
399 
400  // Check whether the polytope of `simplex` is empty. If so, it is trivially
401  // redundant.
402  if (simplex.isEmpty()) {
403  isRedundant[i] = true;
404  continue;
405  }
406 
407  // Check whether `IntegerPolyhedron[i]` is contained in any Poly, that is
408  // different from itself and not yet marked as redundant.
409  for (unsigned j = 0, e = integerPolyhedrons.size(); j < e; ++j) {
410  if (j == i || isRedundant[j])
411  continue;
412 
413  if (simplex.isRationalSubsetOf(integerPolyhedrons[j])) {
414  isRedundant[i] = true;
415  break;
416  }
417  }
418  }
419 
420  for (unsigned i = 0, e = integerPolyhedrons.size(); i < e; ++i)
421  if (!isRedundant[i])
422  newSet.unionPolyInPlace(integerPolyhedrons[i]);
423 
424  return newSet;
425 }
