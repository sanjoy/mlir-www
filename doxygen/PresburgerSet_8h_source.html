20 /// This class can represent a union of IntegerPolyhedrons, with support for
21 /// union, intersection, subtraction and complement operations, as well as
22 /// sampling.
23 ///
24 /// The IntegerPolyhedrons (Polys) are stored in a vector, and the set
25 /// represents the union of these Polys. An empty list corresponds to the empty
26 /// set.
27 ///
28 /// Note that there are no invariants guaranteed on the list of Poly other than
29 /// that they are all in the same space, i.e., they all have the same number of
30 /// dimensions and symbols. For example, the Polys may overlap each other.
31 class PresburgerSet {
32 public:
33  explicit PresburgerSet(const IntegerPolyhedron &poly);
34

36  unsigned getNumPolys() const;
37

39  unsigned getNumDimIds() const;
40 
41  /// Return the number of symbolic dimensions.
42  unsigned getNumSymbolIds() const;
43

45  ArrayRef<IntegerPolyhedron> getAllIntegerPolyhedron() const;
46 
47  /// Return the IntegerPolyhedron at the specified index.
48  const IntegerPolyhedron &getIntegerPolyhedron(unsigned index) const;
49 
50  /// Mutate this set, turning it into the union of this set and the given
51  /// IntegerPolyhedron.
52  void unionPolyInPlace(const IntegerPolyhedron &poly);
53 
54  /// Mutate this set, turning it into the union of this set and the given set.
55  void unionSetInPlace(const PresburgerSet &set);
56 
57  /// Return the union of this set and the given set.
58  PresburgerSet unionSet(const PresburgerSet &set) const;
59 
60  /// Return the intersection of this set and the given set.
61  PresburgerSet intersect(const PresburgerSet &set) const;
62 
63  /// Return true if the set contains the given point, and false otherwise.
64  bool containsPoint(ArrayRef<int64_t> point) const;
65 
66  /// Print the set's internal state.
67  void print(raw_ostream &os) const;
68  void dump() const;
69 
70  /// Return the complement of this set. All local variables in the set must
71  /// correspond to floor divisions.
72  PresburgerSet complement() const;
73 
74  /// Return the set difference of this set and the given set, i.e.,
75  /// return `this \ set`. All local variables in `set` must correspond
76  /// to floor divisions, but local variables in `this` need not correspond to
77  /// divisions.
78  PresburgerSet subtract(const PresburgerSet &set) const;
79 
80  /// Return true if this set is equal to the given set, and false otherwise.
81  /// All local variables in both sets must correspond to floor divisions.
82  bool isEqual(const PresburgerSet &set) const;
83 
84  /// Return a universe set of the specified type that contains all points.
85  static PresburgerSet getUniverse(unsigned numDims = 0,
86  unsigned numSymbols = 0);
87  /// Return an empty set of the specified type that contains no points.
88  static PresburgerSet getEmptySet(unsigned numDims = 0,
89  unsigned numSymbols = 0);
90 
91  /// Return true if all the sets in the union are known to be integer empty
92  /// false otherwise.
93  bool isIntegerEmpty() const;
94 
95  /// Find an integer sample from the given set. This should not be called if
96  /// any of the Polys in the union are unbounded.
97  bool findIntegerSample(SmallVectorImpl<int64_t> &sample);
98 
99  /// Simplifies the representation of a PresburgerSet.
100  ///
101  /// In particular, removes all Polys which are subsets of other Polys in the
102  /// union.
103  PresburgerSet coalesce() const;
104 
105 private:
106  /// Construct an empty PresburgerSet.
107  PresburgerSet(unsigned numDims = 0, unsigned numSymbols = 0)
108  : numDims(numDims), numSymbols(numSymbols) {}
109 
110  /// Return the set difference poly \ set.
111  static PresburgerSet getSetDifference(IntegerPolyhedron poly,
112  const PresburgerSet &set);
113 
114  /// Number of identifiers corresponding to real dimensions.
115  unsigned numDims;
116 
117  /// Number of symbolic dimensions, unknown but constant for analysis, as in
118  /// IntegerPolyhedron.
119  unsigned numSymbols;
120 
121  /// The list of integerPolyhedrons that this set is the union of.
122  SmallVector<IntegerPolyhedron, 2> integerPolyhedrons;
123 };
