49  LogicalResult bufferize(Operation *op, RewriterBase &rewriter,
50  const BufferizationState &state) const {
51  auto selectOp = cast<SelectOp>(op);
52  // `getBuffer` introduces copies if an OpOperand bufferizes out-of-place.
53  // TODO: It would be more efficient to copy the result of the `select` op
54  // instead of its OpOperands. In the worst case, 2 copies are inserted at
55  // the moment (one for each tensor). When copying the op result, only one
56  // copy would be needed.
57  Value trueBuffer =
58  *state.getBuffer(rewriter, selectOp->getOpOperand(1) /*true_value*/);
59  Value falseBuffer =
60  *state.getBuffer(rewriter, selectOp->getOpOperand(2) /*false_value*/);
61  replaceOpWithNewBufferizedOp<SelectOp>(
62  rewriter, op, selectOp.getCondition(), trueBuffer, falseBuffer);
63  return success();
64  }

67  const BufferizationState &state) const {
68  return BufferRelation::None;
69  }
